HTML — هيكلة الموقع

الهدف هنا هو تعلم كيفية بناء الهيكل العظمي لأي صفحة ويب.

1. الأساسيات — كيف يعمل الويب

عندما تكتب عنوان موقع في المتصفح، يرسل طلباً إلى خادم (server) يعيد ملف HTML. المتصفح يقرأ هذا الملف ويعرضه كصفحة ويب.

<!DOCTYPE html>
<html lang="ar">
  <head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>أول صفحة لي</title>
  </head>
  <body>
    <h1>مرحباً بالعالم!</h1>
    <p>هذه أول صفحة ويب لي.</p>
  </body>
</html>
  • <!DOCTYPE html> — تعريف نوع المستند
  • <html> — العنصر الجذر
  • <head> — معلومات تعريفية (ترميز، عنوان، روابط)
  • <body> — المحتوى المرئي

2. العناصر والنصوص

العناوين والفقرات وتنسيق النصوص هي أساس أي صفحة.

<h1>العنوان الأكبر</h1>
<h2>عنوان فرعي</h2>
<h3>عنوان أصغر</h3>
<p>فقرة نصية عادية.</p>
<p><strong>نص عريض</strong> و <em>نص مائل</em>.</p>
<p><u>نص تحته خط</u> و <s>نص مشطوب</s>.</p>
  • h1 إلى h6 — العناوين (h1 الأكبر والأهم)
  • p — الفقرة النصية
  • strong — نص عريض (semantic)
  • em — نص مائل (semantic)
  • br — فاصل سطر
  • hr — خط أفقي فاصل

3. الروابط والوسائط

اربط صفحاتك بالإنترنت وأضف الصور والفيديو.

<!-- الروابط -->
<a href="https://example.com">زيارة الموقع</a>
<a href="/about">عن الموقع</a>
<a href="mailto:email@example.com">راسلنا</a>

<!-- الصور -->
<img src="photo.jpg" alt="وصف الصورة" width="300">

<!-- الفيديو والصوت -->
<video src="video.mp4" controls></video>
<audio src="audio.mp3" controls></audio>

<!-- iframe (تضمين محتوى خارجي) -->
<iframe src="https://www.youtube.com/embed/..."></iframe>
  • a — رابط (target="_blank" لفتح في نافذة جديدة)
  • img — صورة (alt مهم جداً للوصولية)
  • video / audio — وسائط متعددة
  • figure / figcaption — وسائط مع تسمية توضيحية

4. القوائم والجداول

نظّم المعلومات باستخدام القوائم والجداول.

<!-- القوائم غير المرتبة -->
<ul>
  <li>عنصر 1</li>
  <li>عنصر 2</li>
</ul>

<!-- القوائم المرتبة -->
<ol>
  <li>الخطوة الأولى</li>
  <li>الخطوة الثانية</li>
</ol>

<!-- الجداول -->
<table>
  <thead>
    <tr><th>الاسم</th><th>العمر</th></tr>
  </thead>
  <tbody>
    <tr><td>أحمد</td><td>25</td></tr>
  </tbody>
</table>
  • ul — قائمة غير مرتبة (نقاط)
  • ol — قائمة مرتبة (أرقام)
  • table — جدول (thead, tbody, tr, th, td)
  • dl / dt / dd — قائمة تعريفية

5. النماذج والمدخلات

اجمع بيانات المستخدمين باستخدام النماذج.

<form action="/submit" method="POST">
  <label for="name">الاسم:</label>
  <input type="text" id="name" name="name" required>

  <label for="email">البريد:</label>
  <input type="email" id="email" name="email">

  <label for="pass">كلمة المرور:</label>
  <input type="password" id="pass" name="pass">

  <label>الجنس:</label>
  <input type="radio" name="gender" value="male"> ذكر
  <input type="radio" name="gender" value="female"> أنثى

  <label>الاهتمامات:</label>
  <input type="checkbox" name="hobby" value="sports"> رياضة

  <label>الرسالة:</label>
  <textarea name="message" rows="4"></textarea>

  <select name="city">
    <option value="cairo">القاهرة</option>
    <option value="riyadh">الرياض</option>
  </select>

  <button type="submit">إرسال</button>
</form>
  • input[type="text"] — نص عادي
  • input[type="email"] — بريد إلكتروني (تحقق تلقائي)
  • input[type="password"] — كلمة مرور
  • input[type="radio"] — اختيار واحد من مجموعة
  • input[type="checkbox"] — اختيار متعدد
  • textarea — نص طويل
  • select / option — قائمة منسدلة
  • button — زر إرسال
  • required — حقل إجباري
  • placeholder — نص تلميحي داخل الحقل

6. HTML5 الدلالي (Semantic HTML)

استخدم العناصر الدلالية لتحسين محركات البحث (SEO) وإتاحة الوصول (Accessibility).

<header>
  <nav>
    <ul>
      <li><a href="/">الرئيسية</a></li>
      <li><a href="/about">عن الموقع</a></li>
    </ul>
  </nav>
</header>

<main>
  <section>
    <h2>أحدث المقالات</h2>
    <article>
      <h3>عنوان المقال</h3>
      <p>محتوى المقال...</p>
    </article>
  </section>

  <aside>
    <h3>إعلانات</h3>
    <p>محتوى جانبي...</p>
  </aside>
</main>

<footer>
  <p>جميع الحقوق محفوظة &copy; 2026</p>
</footer>
  • header — رأس الصفحة (الشعار والقائمة)
  • nav — روابط التنقل
  • main — المحتوى الرئيسي (مرة واحدة فقط)
  • section — قسم منطقي من المحتوى
  • article — محتوى مستقل بذاته (مقال، خبر)
  • aside — محتوى جانبي (شريط جانبي)
  • footer — تذييل الصفحة
  • figure / figcaption — وسائط مع وصف

CSS — التصميم والتنسيق

حوّل الهيكل الممل إلى واجهة جميلة وجذابة.

1. الأساسيات والمحددات

ثلاث طرق لإضافة CSS والمحددات الأساسية.

/* External (أفضل طريقة) */
/* في ملف style.css */
h1 {
  color: blue;
  font-size: 24px;
}

/* Internal (في نفس الملف داخل <style>) */
/* Inline (على نفس العنصر) */
<p style="color: red;">نص أحمر</p>

/* المحددات */
h1        { }  /* Element - كل العناوين */
.my-class { }  /* Class - مجموعة عناصر */
#my-id    { }  /* ID - عنصر واحد فقط */
*         { }  /* Universal - كل العناصر */

/* محددات متقدمة */
div > p   { }  /* Child - ابن مباشر */
div + p   { }  /* Adjacent sibling */
a:hover   { }  /* Pseudo-class */
p::before { }  /* Pseudo-element */
  • External — <link rel="stylesheet" href="style.css">
  • Internal — <style> في head
  • Inline — style="property: value"
  • الأولوية (Specificity): ID > Class > Element

2. نموذج الصندوق (Box Model)

كل عنصر في CSS هو صندوق. فهم نموذج الصندوق ضروري للتخطيط.

.box {
  width: 300px;
  height: 200px;
  padding: 20px;    /* مسافة داخلية */
  border: 2px solid #333;  /* حدود */
  margin: 15px;     /* مسافة خارجية */
  box-sizing: border-box; /* المهم! */
}
  • Content — المحتوى الفعلي (نص أو صورة)
  • Padding — مسافة داخلية حول المحتوى
  • Border — حدود حول الـ padding
  • Margin — مسافة خارجية حول العنصر
  • box-sizing: border-box — يجعل العرض يشمل padding + border

3. الألوان والخلفيات

أنظمة الألوان المختلفة والخلفيات المتنوعة.

/* أنظمة الألوان */
color: red;              /* اسم لون */
color: #ff0000;          /* HEX */
color: rgb(255, 0, 0);   /* RGB */
color: rgba(255, 0, 0, 0.5); /* RGB مع شفافية */
color: hsl(0, 100%, 50%); /* HSL */

/* الخلفيات */
background-color: #f0f0f0;
background-image: url("bg.jpg");
background-size: cover;
background-position: center;
background-repeat: no-repeat;

/* التدرجات (Gradients) */
background: linear-gradient(to right, red, blue);
background: radial-gradient(circle, red, yellow);
  • HEX — #RRGGBB (الأكثر شيوعاً)
  • RGB — الأحمر الأخضر الأزرق (0-255)
  • HSL — Hue, Saturation, Lightness
  • Gradients — تدرجات خطية وشعاعية

4. تنسيق النصوص والخطوط

تحكم كامل في مظهر النصوص.

/* الخطوط */
font-family: 'Cairo', sans-serif;
font-size: 16px;
font-weight: 700;  /* 400 عادي, 700 عريض */
font-style: italic;

/* المحاذاة والمسافات */
text-align: center;  /* left, right, justify */
line-height: 1.8;    /* ارتفاع السطر */
letter-spacing: 1px; /* مسافة بين الحروف */
word-spacing: 2px;   /* مسافة بين الكلمات */
text-decoration: underline; /* أو none */
text-transform: uppercase;  /* uppercase, lowercase */

/* الظل */
text-shadow: 2px 2px 4px rgba(0,0,0,0.3);
  • Google Fonts — @import url(...) أو <link>
  • Web Safe Fonts — Arial, Verdana, Tahoma, Times New Roman
  • font-weight — سمك الخط (100-900)

5. Flexbox — تخطيط أحادي البعد

نظام مرن لترتيب العناصر في اتجاه واحد.

.container {
  display: flex;
  flex-direction: row;        /* row, column */
  justify-content: center;    /* main axis */
  align-items: center;        /* cross axis */
  flex-wrap: wrap;
  gap: 16px;
}

/* خصائص العناصر */
.item {
  flex: 1;            /* grow, shrink, basis */
  align-self: center; /* override for one item */
  order: 2;           /* change order */
}

خاصيات الحاوية:

  • flex-direction — اتجاه المحور الرئيسي (row/column)
  • justify-content — توزيع العناصر على المحور الرئيسي
  • align-items — محاذاة العناصر على المحور المعاكس
  • flex-wrap — السماح بالتفاف العناصر
  • gap — المسافة بين العناصر

6. CSS Grid — تخطيط ثنائي البعد

نظام شبكي متقدم للتحكم في الصفوف والأعمدة.

.container {
  display: grid;
  grid-template-columns: repeat(3, 1fr);
  grid-template-rows: auto 200px auto;
  gap: 16px;
}

/* عناصر تمتد على أعمدة متعددة */
.item-wide {
  grid-column: span 2;
}

/* تسمية المناطق */
.container {
  grid-template-areas:
    "header header header"
    "sidebar main main"
    "footer footer footer";
}

.header { grid-area: header; }
  • grid-template-columns — تعريف الأعمدة
  • grid-template-rows — تعريف الصفوف
  • gap — المسافة بين الخلايا
  • fr — وحدة كسرية (fraction)
  • grid-area — تسمية المناطق للتخطيط
  • place-items — مختصر justify-items + align-items

7. التصميم المتجاوب (Responsive)

اجعل موقعك يبدو رائعاً على جميع الشاشات.

/* وحدات نسبية */
width: 50%;          /* نسبة من العنصر الأب */
font-size: 1.2rem;   /* نسبة من حجم خط الجذر */
padding: 2em;        /* نسبة من حجم خط العنصر */
width: 100vw;        /* نسبة من عرض الشاشة */
height: 100vh;       /* نسبة من ارتفاع الشاشة */

/* Media Queries */
/* Mobile First - القاعدة للجوال */
.container {
  display: flex;
  flex-direction: column;
}

/* Tablet */
@media (min-width: 768px) {
  .container {
    flex-direction: row;
  }
}

/* Desktop */
@media (min-width: 1024px) {
  .container {
    max-width: 960px;
    margin: 0 auto;
  }
}
  • Mobile First — صمم للجوال أولاً ثم أضف تحسينات
  • Media Queries — نقاط التوقف حسب عرض الشاشة
  • وحدات مرنة — % , rem , em , vw , vh
  • min-width — عند التصغير للأكبر
  • max-width — عند التصغير للأصغر

8. الحركات والتأثيرات (Animations & Transitions)

أضف حركات سلسة وتفاعلية لعناصر صفحتك.

/* Transition - حركة عند التغيير */
.button {
  background-color: blue;
  transition: background-color 0.3s ease;
}
.button:hover {
  background-color: darkblue;
}

/* Animation - حركة مستمرة أو محددة */
@keyframes slide {
  from { transform: translateX(0); }
  to { transform: translateX(100px); }
}

.element {
  animation: slide 1s ease infinite;
}

/* خصائص Transition */
transition: property duration timing-function delay;
/* timing-function: ease, linear, ease-in, ease-out */

/* خصائص Animation */
animation: name duration timing-function delay iteration-count direction;
  • transition — تغيير تدريجي بين حالتين
  • @keyframes — تعريف مراحل الحركة
  • transform — تغيير الشكل (translate, rotate, scale, skew)
  • hover — مرور الفأرة
  • opacity — الشفافية (0 إلى 1)

JavaScript — التفاعل والمنطق

المحرك الأساسي الذي يمنح الحياة للموقع ويجعله تفاعلياً.

1. المتغيرات وأنواع البيانات

أساس البرمجة: خزن البيانات وتعامل معها.

// المتغيرات
let name = "أحمد";         // نص (String)
const PI = 3.14159;        // رقم (Number)
let isActive = true;       // منطقي (Boolean)
let items = ["a", "b"];    // مصفوفة (Array)
let user = { id: 1, name: "Ahmed" }; // كائن (Object)
let x = null;              // فارغ
let y;                     // undefined

// let vs const
let counter = 0;    // يمكن تغييره
counter = 1;

const birthYear = 1990;  // ثابت لا يتغير

// التحويل بين الأنواع
let num = Number("42");   // نص إلى رقم
let str = String(123);    // رقم إلى نص
let bool = Boolean(1);    // رقم إلى منطقي
  • let — متغير قابل للتغيير
  • const — ثابت لا يتغير (مفضل)
  • typeof — معرفة نوع المتغير
  • Number() , String() , Boolean() — تحويل الأنواع

2. التحكم في التدفق (Control Flow)

اتخذ قرارات وكرر العمليات.

// if / else if / else
let age = 18;
if (age >= 18) {
  console.log("بالغ");
} else if (age >= 13) {
  console.log("مراهق");
} else {
  console.log("طفل");
}

// switch
let day = 3;
switch (day) {
  case 1: console.log("الأحد"); break;
  case 2: console.log("الاثنين"); break;
  default: console.log("يوم آخر");
}

// الحلقات التكرارية
// for
for (let i = 0; i < 5; i++) {
  console.log(i);
}

// while
let i = 0;
while (i < 5) {
  console.log(i);
  i++;
}

// do while
let j = 0;
do {
  console.log(j);
  j++;
} while (j < 5);

// for...of (للمصفوفات)
for (let item of items) { ... }

// for...in (للكائنات)
for (let key in user) { ... }

// المقارنة
==    // يساوي (مع تحويل النوع)
===   // يساوي تماماً (نفس القيمة والنوع)
!=    // لا يساوي
!==   // لا يساوي تماماً
> < >= <=  // مقارنات رقمية

// المنطق
&&   // AND
||    // OR
!     // NOT

3. الدوال (Functions)

أعد استخدام الكود و نظم منطقك.

// دالة تقليدية
function greet(name) {
  return "مرحباً " + name + "!";
}

// Arrow function (ES6)
const greet = (name) => `مرحباً ${name}!`;

// مع قيمة افتراضية
function multiply(a, b = 1) {
  return a * b;
}

// دالة تأخذ دالة أخرى (Callback)
function process(callback) {
  const data = "معلومات";
  callback(data);
}

process((data) => console.log(data));

// Parameters vs Arguments
function sayHello(name) { }  // name = parameter
sayHello("Ahmed");           // "Ahmed" = argument

// Return
function add(a, b) {
  return a + b;  // ترجع القيمة
  console.log("لن يظهر");  // لا ينفذ
}
  • Function declaration — function name() {}
  • Arrow function — const name = () => {}
  • Parameters — المدخلات
  • Return — القيمة المرجعة
  • Scope — المتغيرات داخل الدالة لا تظهر خارجها

4. المصفوفات والكائنات

خزن البيانات المنظمة وتعامل معها بطرق حديثة.

// المصفوفات
let fruits = ["تفاح", "موز", "برتقال"];
fruits.push("عنب");         // إضافة في النهاية
fruits.pop();               // حذف من النهاية
fruits.unshift("فراولة");   // إضافة في البداية
fruits.shift();             // حذف من البداية

// طرق المصفوفات الحديثة
let numbers = [1, 2, 3, 4, 5];

// map - تحويل كل عنصر
let doubled = numbers.map(n => n * 2);
// [2, 4, 6, 8, 10]

// filter - تصفية
let evens = numbers.filter(n => n % 2 === 0);
// [2, 4]

// forEach - التنفيذ على كل عنصر
numbers.forEach(n => console.log(n));

// find - إيجاد أول عنصر يحقق الشرط
let found = numbers.find(n => n > 3);
// 4

// reduce - تجميع القيم
let sum = numbers.reduce((acc, n) => acc + n, 0);
// 15

// الكائنات
let person = {
  name: "أحمد",
  age: 25,
  greet() {
    console.log(`مرحباً، أنا ${this.name}`);
  }
};

person.name;          // "أحمد"
person["age"];        // 25
person.greet();       // "مرحباً، أنا أحمد"

parent.insertBefore(newDiv, referenceChild);

6. الأحداث (Events)

تفاعل مع المستخدم: النقر، الكتابة، تحريك الفأرة.

// النقر
button.addEventListener("click", (event) => {
  console.log("تم النقر!", event);
});

// لوحة المفاتيح
document.addEventListener("keydown", (e) => {
  if (e.key === "Enter") {
    console.log("ضغط Enter");
  }
});

// إرسال النموذج
form.addEventListener("submit", (e) => {
  e.preventDefault();  // منع إعادة تحميل الصفحة
  console.log("تم الإرسال");
});

// الماوس
element.addEventListener("mouseenter", () => { });
element.addEventListener("mouseleave", () => { });

// المدخلات
input.addEventListener("input", (e) => {
  console.log(e.target.value);
});

// أحداث شائعة
// click, dblclick
// mouseenter, mouseleave, mousemove
// keydown, keyup
// submit, change, input, focus, blur
// scroll, resize, load, DOMContentLoaded

// إزالة الحدث
button.removeEventListener("click", handler);
  • addEventListener — أفضل طريقة لإضافة الأحداث
  • event.preventDefault() — منع السلوك الافتراضي
  • event.target — العنصر الذي أثار الحدث
  • event.key — المفتاح المضغوط (للوحة المفاتيح)
  • DOMContentLoaded — يشغل الكود بعد تحميل HTML

API — جلب البيانات من الإنترنت

اربط تطبيقك بالعالم الخارجي وابنِ تطبيقات كاملة.

1. ما هو API؟

API (Application Programming Interface) هو مجموعة من القواعد التي تسمح لتطبيقين بالتواصل. في تطوير الويب، APIs تتيح للواجهة الأمامية التحدث مع الخادم الخلفي.

  • Web API — نقاط نهاية HTTP ترجع بيانات (غالباً JSON)
  • REST API — النمط المعماري الأكثر شيوعاً
  • Third-party APIs — خدمات مثل GitHub و Twitter و Stripe
  • API Key — مفتاح للمصادقة على الطلبات
  • Endpoint — عنوان URL محدد لجلب أو إرسال بيانات

2. REST و HTTP Methods

REST (Representational State Transfer) هو أشهر نمط لبناء APIs. كل مورد (resource) له URL.

GETقراءة / جلب بيانات/users قائمة المستخدمين
POSTإنشاء بيانات جديدة/users إضافة مستخدم
PUTتحديث كامل/users/1 تحديث المستخدم
PATCHتحديث جزئي/users/1 تحديث الاسم فقط
DELETEحذف/users/1 حذف المستخدم

حالات HTTP: 200 OK, 201 Created, 400 Bad Request, 401 Unauthorized, 404 Not Found, 500 Internal Server Error.

3. JSON (JavaScript Object Notation)

JSON هو الصيغة القياسية لتبادل البيانات عبر APIs. يشبه كائنات JavaScript لكن مع علامات اقتباس مزدوجة.

// بيانات JSON — ما ترجعه API
{
  "userId": 1,
  "id": 1,
  "title": "تعلم APIs",
  "completed": false
}

// تحويل JSON في JavaScript
const jsonString = JSON.stringify({ name: "أحمد" });
// "{"name":"أحمد"}"

const obj = JSON.parse(jsonString);
// { name: "أحمد" }
  • JSON.stringify() — كائن JavaScript → نص JSON
  • JSON.parse() — نص JSON → كائن JavaScript
  • JSON يدعم: strings, numbers, objects, arrays, booleans, null

4. Fetch API — طلبات GET

Fetch هي واجهة برمجية حديثة لعمل طلبات HTTP من المتصفح.

// GET — جلب البيانات
fetch("https://jsonplaceholder.typicode.com/posts")
  .then(response => {
    if (!response.ok) {
      throw new Error(`HTTP error! status: ${response.status}`);
    }
    return response.json();
  })
  .then(posts => {
    console.log("المنشورات:", posts);
    posts.forEach(post => {
      console.log(post.title);
    });
  })
  .catch(error => {
    console.error("فشل الطلب:", error);
  });

// GET مع معاملات (query parameters)
fetch("https://api.example.com/users?page=1&limit=10")
  .then(res => res.json())
  .then(data => console.log(data));
  • fetch(url) — يبدأ الطلب
  • response.ok — تحقق من نجاح الطلب (status 200-299)
  • response.json() — تحويل الرد إلى كائن JavaScript
  • .catch() — معالجة أخطاء الشبكة

5. Fetch API — طلبات POST

أرسل بيانات جديدة إلى الخادم.

// POST — إنشاء بيانات جديدة
const newPost = {
  title: "مقال جديد",
  body: "هذا هو محتوى المقال.",
  userId: 1
};

fetch("https://jsonplaceholder.typicode.com/posts", {
  method: "POST",
  headers: {
    "Content-Type": "application/json"
  },
  body: JSON.stringify(newPost)
})
  .then(response => {
    if (!response.ok) throw new Error("فشل الإرسال");
    return response.json();
  })
  .then(createdPost => {
    console.log("تم الإنشاء:", createdPost);
  })
  .catch(error => {
    console.error("خطأ:", error);
  });

// PUT — تحديث كامل
fetch("https://jsonplaceholder.typicode.com/posts/1", {
  method: "PUT",
  headers: { "Content-Type": "application/json" },
  body: JSON.stringify({ title: "مُحدّث", body: "محتوى جديد", userId: 1 })
});

// DELETE — حذف
fetch("https://jsonplaceholder.typicode.com/posts/1", {
  method: "DELETE"
});

6. Async / Await وأفضل الممارسات

اكتب كوداً غير متزامن يبدو وكأنه متزامن — أنظف وأسهل للقراءة.

// async/await
async function getUsers() {
  try {
    const response = await fetch("https://jsonplaceholder.typicode.com/users");
    if (!response.ok) throw new Error("فشل الطلب");
    const users = await response.json();
    return users;
  } catch (error) {
    console.error("خطأ:", error);
    return [];
  }
}

// استخدام الدالة
getUsers().then(users => console.log(users));

// طلبات متوازية (أسرع)
async function getDashboardData() {
  try {
    const [users, posts, comments] = await Promise.all([
      fetch("/api/users").then(r => r.json()),
      fetch("/api/posts").then(r => r.json()),
      fetch("/api/comments").then(r => r.json())
    ]);
    return { users, posts, comments };
  } catch (error) {
    console.error("فشل تحميل البيانات:", error);
  }
}

/**
 * أفضل الممارسات:
 * 1. استخدم async/await بدلاً من .then()
 * 2. تعامل مع الأخطاء بـ try/catch
 * 3. أظهر حالة التحميل للمستخدم
 * 4. تحقق من response.ok قبل قراءة البيانات
 * 5. استخدم Promise.all للطلبات المتوازية
 * 6. لا تنسى إلغاء الاشتراك (abort) عند مغادرة الصفحة
 */
  • async — تحول الدالة إلى غير متزامنة (ترجع Promise)
  • await — ينتظر حتى ينتهي الـ Promise
  • try/catch — معالجة الأخطاء في الدوال غير المتزامنة
  • Promise.all — تنفيذ طلبات متعددة بالتوازي
  • AbortController — إلغاء الطلب عند الحاجة

مثال تطبيقي كامل

تطبيق بسيط يجلب بيانات مستخدم من GitHub ويعرضها:

async function showGitHubUser(username) {
  const output = document.getElementById("result");

  // 1. حالة التحميل
  output.innerHTML = "<p>جاري التحميل...</p>";

  try {
    // 2. طلب API
    const res = await fetch(`https://api.github.com/users/${username}`);

    if (!res.ok) {
      throw new Error(res.status === 404
        ? "المستخدم غير موجود"
        : "حدث خطأ في الخادم");
    }

    const data = await res.json();

    // 3. عرض النتيجة
    output.innerHTML = `
      <div class="user-card">
        <img src="${data.avatar_url}" alt="${data.login}" width="80">
        <h3>${data.name || data.login}</h3>
        <p>${data.bio || "لا يوجد وصف"}</p>
        <p>📍 ${data.location || "غير محدد"}</p>
        <p>📦 ${data.public_repos} مستودع</p>
      </div>
    `;

  } catch (error) {
    // 4. معالجة الأخطاء
    output.innerHTML = `<p class="error">❌ ${error.message}</p>`;
  }
}

// الاستخدام
showGitHubUser("octocat");
